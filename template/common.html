Sum of 1 to N number 


public class sum
 {
     public static void main(String[] args)
     {
         java.util.Scanner scn = new java.util.Scanner (System.in);
         System.out.println("Entre the valur of N ");
         int n= scn.nextInt();
         int sum=0;
         for(int i=1;i<=n;i++)
         {
             sum =sum+i;
         }
         System.out.println("The sum of 1st "+n+"= " +sum);
     }
 }
