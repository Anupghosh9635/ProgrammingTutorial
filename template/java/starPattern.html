public class StarTrangle
{
    public static void main(String[] args)
    {
        java.util.Scanner sc=new java.util.Scanner(System.in);
        System.out.println("Enter the value of * printnting line no>- ");
        int n =sc.nextInt();
        int i,j;
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=(2*n-1);j++)
            {
                if(i+j<=n || i+j>=n+2*i)
                {
                    System.out.print(" ");
                }
                else
                    System.out.print("*");
            }
            System.out.println();
        }
    }
}

public class StarDimond
{
    public static void main(String[] args)
    {
        java.util.Scanner scn=new java.util.Scanner(System.in);
        System.out.println(" Enter the value to * print >- ");
        int n=scn.nextInt();
        int str=1;
        int spa=n/2;
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=spa;j++)
            {
                System.out.print(" ");
            }
            for (int k=1;k<=str;k++ )
            {
                System.out.print("*");
            }
            System.out.println();
            if(i<=n/2)
            {
                spa--;
                str+=2;
            }else
            {
                spa++;
                str-=2;
            }
        }
    }
}
